<script>
/*
  Vitium Veritas™ Admin Console Script
  -----------------------------------
  This script provides:
  1. Administrator PIN verification for unlocking the admin console.
  2. Keypad interaction for numeric input.
  3. Conditional display of the admin dashboard after successful unlock.
  4. Copilot assistance panel toggle behaviour (demo only, no live backend).
*/

(function () {
  // Elements related to PIN unlock
  const pinInput = document.getElementById('vv-pin-input');
  const keypadButtons = document.querySelectorAll('.vv-keypad button');
  const unlockButton = document.getElementById('vv-unlock-btn');
  const adminPinCard = document.getElementById('vv-admin-pin-card');
  const adminDashboard = document.getElementById('vv-admin-dashboard');

  // Elements related to Copilot assistance
  const copilotToggleContainer = document.getElementById('vv-copilot-toggle');
  const copilotOpenButton = document.getElementById('vv-copilot-open-btn');
  const copilotPanel = document.getElementById('vv-copilot-panel');
  const copilotCloseButton = document.getElementById('vv-copilot-close-btn');
  const copilotChips = document.querySelectorAll('.vv-copilot-chip');
  const copilotInput = document.getElementById('vv-copilot-input');

  /**
   * Append a digit to the PIN input, respecting the maximum length.
   * @param {string} digit
   */
  function appendDigit(digit) {
    if (pinInput.value.length < pinInput.maxLength) {
      pinInput.value += digit;
    }
  }

  /**
   * Remove the last character from the PIN input.
   */
  function backspace() {
    pinInput.value = pinInput.value.slice(0, -1);
  }

  /**
   * Clear the PIN input field.
   */
  function clearInput() {
    pinInput.value = '';
  }

  /**
   * Attempt to unlock the admin console using the current PIN value.
   */
  function unlock() {
    const pin = pinInput.value;
    verifyPIN(pin);
  }

  /**
   * Verify the provided PIN.
   * In this demo, the PIN is hard coded. In production, this must be validated server side.
   * @param {string} pin
   */
  function verifyPIN(pin) {
    // Demo PIN: 1937 (same as dashboard.html for consistency in the demo environment)
    if (pin === '1937') {
      // Hide the PIN card and reveal the admin dashboard
      adminPinCard.style.display = 'none';
      adminDashboard.style.display = 'block';

      // Show Copilot toggle for admin only assistance
      copilotToggleContainer.style.display = 'block';
    } else {
      alert('Invalid administrator PIN');
    }
  }

  // Attach keypad button behaviour
  keypadButtons.forEach(function (btn) {
    btn.addEventListener('click', function () {
      const key = btn.dataset.key;

      if (key === 'back') {
        backspace();
        return;
      }
      if (key === 'clear') {
        clearInput();
        return;
      }

      appendDigit(key);
    });
  });

  // Attach unlock button behaviour
  unlockButton.addEventListener('click', function () {
    unlock();
  });

  // Allow Enter key to trigger unlock
  pinInput.addEventListener('keydown', function (event) {
    if (event.key === 'Enter') {
      event.preventDefault();
      unlock();
    }
  });

  /**
   * Open the Copilot assistance panel.
   */
  function openCopilotPanel() {
    copilotPanel.classList.add('open');
  }

  /**
   * Close the Copilot assistance panel.
   */
  function closeCopilotPanel() {
    copilotPanel.classList.remove('open');
  }

  // Copilot toggle button behaviour
  copilotOpenButton.addEventListener('click', function () {
    openCopilotPanel();
  });

  // Copilot close button behaviour
  copilotCloseButton.addEventListener('click', function () {
    closeCopilotPanel();
  });

  // Allow clicking on example chips to populate the Copilot input
  copilotChips.forEach(function (chip) {
    chip.addEventListener('click', function () {
      copilotInput.value = chip.textContent.trim();
      copilotInput.focus();
    });
  });

})();
</script>
